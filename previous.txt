from fastapi import FastAPI, UploadFile, File
from fastapi.middleware.cors import CORSMiddleware
from pypdf import PdfReader
import io
import requests
import re

app = FastAPI()

# 🔥 IMPORTANT: allows frontend to talk to backend
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

@app.post("/match")
async def match(file: UploadFile = File(...)):
    contents = await file.read()

    pdf = PdfReader(io.BytesIO(contents))

    text = ""
    for page in pdf.pages:
        text += page.extract_text() or ""

    # clean text
    text = text.lower()
    text = re.sub(r'\s+', ' ', text)

    # use resume text as search query
    query = text[:200]

    # fetch jobs from Adzuna
    jobs = get_jobs(query)

    # match jobs to resume
    results = []
    for job in jobs:
        job_text = (job["title"] + " " + job["description"]).lower()

        # simple scoring (keyword overlap)
        score = sum(1 for word in query.split() if word in job_text)

        results.append({
            "title": job["title"],
            "company": job["company"],
            "score": score
        })

    # sort by best match
    results = sorted(results, key=lambda x: x["score"], reverse=True)

    return {"matches": results}

from fastapi import FastAPI, UploadFile, File
from fastapi.middleware.cors import CORSMiddleware
from pypdf import PdfReader
import io
import requests
import re
import os

ADZUNA_APP_ID = os.getenv("ADZUNA_APP_ID")
ADZUNA_APP_KEY = os.getenv("ADZUNA_APP_KEY")

app = FastAPI()

# 🔥 IMPORTANT: allows frontend to talk to backend
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

@app.post("/match")
async def match(file: UploadFile = File(...)):
    contents = await file.read()

    pdf = PdfReader(io.BytesIO(contents))

    text = ""
    for page in pdf.pages:
        text += page.extract_text() or ""

    # clean text
    text = text.lower()
    text = re.sub(r'\s+', ' ', text)

    # use resume text as search query
    
    keywords = [w for w in text.split() if len(w) > 3]
    query = " ".join(keywords[:15])

    # fetch jobs from Adzuna
    jobs = get_jobs(query)

    # match jobs to resume
    results = []
    for job in jobs:
        job_text = (job["title"] + " " + job["description"]).lower()

        # simple scoring (keyword overlap)
        query_words = set(query.split())
        job_words = set(job_text.split())

        score = len(query_words.intersection(job_words))

        results.append({
            "title": job["title"],
            "company": job["company"],
            "score": score
        })

    # sort by best match
    results = sorted(results, key=lambda x: x["score"], reverse=True)

    return {"matches": results}



def get_jobs(query):
    url = "https://api.adzuna.com/v1/api/jobs/ca/search/1"

    params = {
        "app_id": ADZUNA_APP_ID,
        "app_key": ADZUNA_APP_KEY,
        "what": query,
        "results_per_page": 10,
        "content-type": "application/json"
    }

    res = requests.get(url, params=params)
    
    if res.status_code != 200:
        return []

    data = res.json()
    
    
    jobs = []
    for item in data.get("results", []):
        jobs.append({
            "title": item.get("title"),
            "company": item.get("company", {}).get("display_name"),
            "description": item.get("description", "")
        })

    return jobs
